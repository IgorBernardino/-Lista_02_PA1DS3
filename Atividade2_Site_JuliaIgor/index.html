01

Soma de números válidos

Enunciado

Leia números inteiros positivos até que o usuário digite um número negativo. Ignore os números iguais a zero, some apenas os positivos e conte quantos números válidos foram digitados. Ao final, mostre a soma total, a quantidade de números válidos e a média dos valores válidos.

Código C++ — JuliaIgor001.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 1 - Soma de numeros validos
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia;          // numero digitado
    double igor = 0;    // soma dos positivos
    int contador = 0;   // quantidade de numeros validos

    cout << "Digite numeros inteiros positivos (negativo encerra):" << endl;
    cin >> julia;

    while (julia >= 0) {
        if (julia > 0) {     // ignora os zeros
            igor += julia;
            contador++;
        }
        cin >> julia;
    }

    cout << "\nSoma total: " << igor << endl;
    cout << "Quantidade de numeros validos: " << contador << endl;
    if (contador > 0)
        cout << "Media dos valores validos: " << igor / contador << endl;
    else
        cout << "Nenhum numero valido digitado." << endl;

    return 0;
}

02

Verificação de número primo

Enunciado

Leia um número inteiro positivo e determine se ele é primo. Um número primo possui exatamente dois divisores: 1 e ele mesmo.

Código C++ — JuliaIgor002.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 2 - Verificacao de numero primo
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia;            // numero a ser testado
    int igor = 0;         // contador de divisores

    cout << "Digite um numero inteiro positivo: ";
    cin >> julia;

    for (int i = 1; i <= julia; i++) {
        if (julia % i == 0)
            igor++;       // encontrou um divisor
    }

    if (julia > 1 && igor == 2)
        cout << "O numero " << julia << " e PRIMO." << endl;
    else
        cout << "O numero " << julia << " NAO e primo." << endl;

    return 0;
}

04

Contagem de números primos em intervalo

Enunciado

Leia dois inteiros positivos A e B. Mostre todos os números primos no intervalo e a quantidade total de primos. Observação: se A > B, não faça nada.

Código C++ — JuliaIgor004.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 4 - Contagem de numeros primos em intervalo
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia;        // limite A
    int igor;         // limite B
    int total = 0;    // quantidade de primos

    cout << "Digite A: ";
    cin >> julia;
    cout << "Digite B: ";
    cin >> igor;

    if (julia <= igor) {     // se A > B nao faz nada
        cout << "\nNumeros primos no intervalo:" << endl;
        for (int n = julia; n <= igor; n++) {
            int divisores = 0;
            for (int i = 1; i <= n; i++) {
                if (n % i == 0)
                    divisores++;
            }
            if (n > 1 && divisores == 2) {
                cout << n << " ";
                total++;
            }
        }
        cout << "\n\nQuantidade total de primos: " << total << endl;
    }

    return 0;
}

Aproximação de π (PI)

Enunciado

Aproxime π usando a série Pi = 4 × (1 − 1/3 + 1/5 − 1/7 + ...). Leia a quantidade de termos da série.

Código C++ — JuliaIgor005.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 5 - Aproximacao de PI
 Pi = 4 * (1 - 1/3 + 1/5 - 1/7 + ...)
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia;            // quantidade de termos
    double igor = 0;      // soma da serie
    int sinal = 1;        // alterna +1 e -1
    int denominador = 1;

    cout << "Digite a quantidade de termos da serie: ";
    cin >> julia;

    for (int i = 0; i < julia; i++) {
        igor += (double)sinal / denominador;
        sinal = -sinal;
        denominador += 2;
    }

    double pi = 4 * igor;
    cout << "\nValor aproximado de PI: " << pi << endl;

    return 0;
}

Padrão avançado com caracteres

Enunciado

Leia um número N e exiba o padrão de asteriscos: de N até 1 (descendo) e de 2 até N (subindo).

Código C++ — JuliaIgor006.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 6 - Padrao avancado com caracteres
 Para N=5: 5 a 1 asteriscos descendo e 2 a 5 subindo
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia;   // valor de N
    int igor;    // quantidade de asteriscos por linha

    cout << "Digite um Numero: ";
    cin >> julia;

    cout << endl;
    // parte que desce: N ate 1
    for (igor = julia; igor >= 1; igor--) {
        for (int j = 1; j <= igor; j++)
            cout << "*";
        cout << endl;
    }
    // parte que sobe: 2 ate N
    for (igor = 2; igor <= julia; igor++) {
        for (int j = 1; j <= igor; j++)
            cout << "*";
        cout << endl;
    }

    return 0;
}

08

Sequência de Collatz

Enunciado

Implemente a sequência de Collatz: se o número for par, divide por 2; se for ímpar, multiplica por 3 e soma 1. Termina ao chegar em 1. Mostre todos os valores e quantos passos foram necessários.

Código C++ — JuliaIgor008.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 8 - Sequencia de Collatz
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia;        // numero inicial
    int igor = 0;     // contador de passos

    cout << "Digite um numero inteiro positivo: ";
    cin >> julia;

    cout << endl << julia << endl;
    while (julia != 1) {
        if (julia % 2 == 0)
            julia = julia / 2;     // par
        else
            julia = julia * 3 + 1; // impar
        cout << julia << endl;
        igor++;
    }

    cout << "Passos = " << igor << endl;

    return 0;
}

Conversão Decimal para Binário

Enunciado

Leia um número decimal positivo e exiba sua representação binária.

Código C++ — JuliaIgor009.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 9 - Conversao Decimal para Binario
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia;            // numero decimal
    int igor[32];         // vetor que guarda os bits
    int qtd = 0;

    cout << "Digite um numero decimal positivo: ";
    cin >> julia;

    int n = julia;
    if (n == 0)
        igor[qtd++] = 0;
    while (n > 0) {
        igor[qtd++] = n % 2;   // resto da divisao por 2
        n = n / 2;
    }

    cout << "Binario: ";
    for (int i = qtd - 1; i >= 0; i--)   // do ultimo para o primeiro
        cout << igor[i];
    cout << endl;

    return 0;
}

Vetor automático com a sequência de Fibonacci

Enunciado

Preencha automaticamente um vetor de 16 posições inteiras com a sequência de Fibonacci (0, 1, 1, 2, 3, 5, ...). Não há digitação de dados: o programa popula e exibe o vetor com 16 elementos.

Código C++ — JuliaIgor011.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 11 - Vetor de 16 posicoes com Fibonacci
 (sem digitacao - preenchimento automatico)
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia[16];   // vetor que recebe a sequencia
    int igor;        // indice de controle

    julia[0] = 0;
    julia[1] = 1;
    for (igor = 2; igor < 16; igor++) {
        julia[igor] = julia[igor - 1] + julia[igor - 2];
    }

    cout << "Sequencia de Fibonacci (16 elementos):" << endl;
    for (igor = 0; igor < 16; igor++) {
        cout << julia[igor] << " ";
    }
    cout << endl;

    return 0;
}

Dois vetores reais (digitado e automático)

Enunciado

Preencha por digitação um vetor real de 20 posições. Em seguida, popule automaticamente um segundo vetor de 20 posições: se o 1º elemento do digitado for igual a zero, copie igual; se for maior que zero, multiplique cada elemento por 0,5; se for menor que zero, divida cada elemento por 0,1.

Código C++ — JuliaIgor012.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 12 - Dois vetores reais (digitado e automatico)
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    double julia[20];   // vetor digitado
    double igor[20];    // vetor preenchido automaticamente

    cout << "Digite 20 numeros reais:" << endl;
    for (int i = 0; i < 20; i++) {
        cout << "Posicao " << i << ": ";
        cin >> julia[i];
    }

    // regra baseada na posicao zero do vetor digitado
    for (int i = 0; i < 20; i++) {
        if (julia[0] == 0)
            igor[i] = julia[i];          // copia identica
        else if (julia[0] > 0)
            igor[i] = julia[i] * 0.5;    // metade
        else
            igor[i] = julia[i] / 0.1;    // dividido por 0,1
    }

    cout << "\nVetor automatico (igor):" << endl;
    for (int i = 0; i < 20; i++) {
        cout << igor[i] << " ";
    }
    cout << endl;

    return 0;
}

 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 13 - Dois vetores inteiros (50 posicoes)
 vetor 1: 1..50   vetor 2: 50..1
*****************************************************/

using namespace std;

int main() {
    int julia[50];   // 1, 2, 3, ... 50

    // 2a estrutura: exibe o primeiro vetor
    cout << "Vetor 1 (crescente):" << endl;
    for (int i = 0; i < 50; i++)
        cout << julia[i] << " ";
    cout << endl;

    // 3a estrutura: exibe o segundo vetor
    cout << "\nVetor 2 (decrescente):" << endl;
    for (int i = 0; i < 50; i++)
        cout << igor[i] << " ";
    cout << endl;

    return 0;
}

 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 14 - Procurar dado num vetor de 10 reais
*****************************************************/

using namespace std;

int main() {
    double julia[10];   // vetor digitado
    double igor;        // numero procurado

        cout << "Posicao " << i << ": ";

    cout << "Digite o numero a procurar: ";
    cin >> igor;

    for (int i = 0; i < 10; i++) {
        if (julia[i] == igor) {
            cout << "Encontrado na posicao (indice) " << i << endl;
            achou = true;
        }
    }

    if (!achou)
        cout << "Numero digitado nao pertence ao vetor" << endl;

    return 0;
}

Matriz 5x5 a partir do número central

Enunciado

Digite o número inteiro que fica no centro da matriz 5×5. Os demais são preenchidos automaticamente, crescendo para a direita e decrescendo para a esquerda.

Código C++ — JuliaIgor018.cpp 
/*****************************************************
 Atividade 2 - 1DS3 - Programacao e Algoritmos
 Dupla: JULIA / IGOR
 Exercicio 18 - Matriz 5x5 a partir do numero central
 (crescente a direita, decrescente a esquerda)
*****************************************************/
#include <iostream>
using namespace std;

int main() {
    int julia[5][5];   // matriz
    int igor;          // numero do centro

    cout << "Digite o numero do centro da matriz 5x5: ";
    cin >> igor;

    // centro = [2][2]. Cada posicao varia conforme o indice linear.
    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 5; j++) {
            int posicao = i * 5 + j;   // 0..24
            julia[i][j] = igor + (posicao - 12);
        }
    }

    cout << "\nMatriz 5x5:" << endl;
    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 5; j++)
            cout << julia[i][j] << "\t";
        cout << endl;
    }

    return 0;
}
